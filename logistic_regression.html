library(broom)

trial <- tibble(
  estrogen = c(rep(c("no", "yes"), 2)), 
  event = c(0, 0, 1, 1), 
  count = c(7479, 7755, 623, 751)
)

trial_mod <- glm(event ~ estrogen, weights = count, data = trial, 
                 family = "binomial")

summary(trial_mod)

library(broom)

low_bwt <- read_csv("../../data/lowbwt.csv")

bwt_mod <- glm(low ~ age, data = low_bwt, family = "binomial")
